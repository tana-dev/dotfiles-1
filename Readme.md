# 導入
## install.sh
$ cd ~

$ git clone https://github.com/tana-dev/dotfiles.git

$ cd dotfiles

$ chmod 777 install.sh

$ ./install.sh

## dein.vimを使う場合
$ mkdir -p ~/.vim/bundle

$ curl https://raw.githubusercontent.com/Shougo/dein.vim/master/bin/installer.sh > installer.sh

$ sh installer.sh ~/.vim/bundle
